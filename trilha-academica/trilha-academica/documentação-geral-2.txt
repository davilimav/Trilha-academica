## 1. VISÃO GERAL DO PROJETO

### 1.1 Objetivo
O **Trilha Acadêmica** é um sistema inteligente de recomendação de trilhas de aprendizado personalizadas. O usuário responde um questionário sobre seu nível de conhecimento em diferentes áreas, e o sistema utiliza **Machine Learning** para recomendar as melhores trilhas de estudo.

### 1.2 Tecnologias Utilizadas

| Camada | Tecnologia | Versão |
|--------|------------|--------|
| **Backend API** | Node.js + Express | 18.x |
| **Banco de Dados** | MySQL | 8.0 |
| **Machine Learning** | Python + Flask + scikit-learn | 3.9+ |
| **Frontend** | HTML5 + CSS3 + JavaScript | - |
| **Autenticação** | JWT (JSON Web Token) | - |

### 4.2 Autenticação (`backend/controllers/authController.js`)

```javascript
const bcrypt = require('bcryptjs');
const jwt = require('jsonwebtoken');
const { pool } = require('../config/database');

// Registrar usuário
const register = async (req, res) => {
    try {
        const { name, email, password } = req.body;
        
        if (!name || !email || !password) {
            return res.status(400).json({ error: 'Todos os campos são obrigatórios' });
        }
        
        const [existing] = await pool.query('SELECT id FROM users WHERE email = ?', [email]);
        if (existing.length > 0) {
            return res.status(400).json({ error: 'Email já cadastrado' });
        }
        
        const hashedPassword = await bcrypt.hash(password, 10);
        const [result] = await pool.query(
            'INSERT INTO users (name, email, password) VALUES (?, ?, ?)',
            [name, email, hashedPassword]
        );
        
        res.status(201).json({ 
            success: true, 
            message: 'Cadastro realizado com sucesso!',
            user: { id: result.insertId, name, email }
        });
    } catch (error) {
        console.error('Register error:', error);
        res.status(500).json({ error: 'Erro interno do servidor' });
    }
};

// Login
const login = async (req, res) => {
    try {
        const { email, password } = req.body;
        
        if (!email || !password) {
            return res.status(400).json({ error: 'Email e senha são obrigatórios' });
        }
        
        const [users] = await pool.query('SELECT * FROM users WHERE email = ?', [email]);
        
        if (users.length === 0) {
            return res.status(401).json({ error: 'Email ou senha inválidos' });
        }
        
        const user = users[0];
        const isValid = await bcrypt.compare(password, user.password);
        
        if (!isValid) {
            return res.status(401).json({ error: 'Email ou senha inválidos' });
        }
        
        const token = jwt.sign(
            { id: user.id, name: user.name, email: user.email },
            process.env.JWT_SECRET,
            { expiresIn: '7d' }
        );
        
        res.json({
            success: true,
            token,
            user: { id: user.id, name: user.name, email: user.email }
        });
    } catch (error) {
        console.error('Login error:', error);
        res.status(500).json({ error: 'Erro no login' });
    }
};

module.exports = { register, login };
```

### 4.3 Middleware de Autenticação (`backend/middlewares/auth.js`)

```javascript
const jwt = require('jsonwebtoken');

const authMiddleware = (req, res, next) => {
    const token = req.headers.authorization?.split(' ')[1];
    
    if (!token) {
        return res.status(401).json({ error: 'Token não fornecido' });
    }
    
    try {
        const decoded = jwt.verify(token, process.env.JWT_SECRET);
        req.user = decoded;
        next();
    } catch {
        res.status(403).json({ error: 'Token inválido' });
    }
};

module.exports = { authMiddleware };
```

const axios = require('axios');

const PYTHON_API_URL = process.env.PYTHON_API_URL || 'http://localhost:5000';

class RecomendacaoService {
    async getPerguntas() {
        try {
            const response = await axios.get(`${PYTHON_API_URL}/questionario/perguntas`);
            return response.data;
        } catch (error) {
            console.error('Erro ao buscar perguntas:', error.message);
            return { perguntas: [] };
        }
    }
    
    async getRecomendacoes(respostas) {
        try {
            const response = await axios.post(`${PYTHON_API_URL}/questionario/recomendar`, respostas);
            return response.data;
        } catch (error) {
            console.error('Erro ao obter recomendações:', error.message);
            return { success: false, error: 'Serviço indisponível', recomendacoes: [] };
        }
    }
    
    async healthCheck() {
        try {
            const response = await axios.get(`${PYTHON_API_URL}/health`);
            return response.data;
        } catch (error) {
            return { status: 'offline' };
        }
    }
}

module.exports = new RecomendacaoService();

        const userId = req.user.id;

            `INSERT INTO questionario_respostas 
             (user_id, programacao_nivel, dados_nivel, design_nivel, seguranca_nivel, marketing_nivel, experiencia_anos, objetivo) 
             VALUES (?, ?, ?, ?, ?, ?, ?, ?)
             ON DUPLICATE KEY UPDATE
             programacao_nivel = VALUES(programacao_nivel),
             dados_nivel = VALUES(dados_nivel),
             design_nivel = VALUES(design_nivel),
             seguranca_nivel = VALUES(seguranca_nivel),
             marketing_nivel = VALUES(marketing_nivel),
             experiencia_anos = VALUES(experiencia_anos),
             objetivo = VALUES(objetivo)`,
            [userId, programacao_nivel, dados_nivel, design_nivel, seguranca_nivel, marketing_nivel, experiencia_anos, objetivo]
        );
        
        res.json({ success: true, message: 'Respostas salvas com sucesso' });
    } catch (error) {
        console.error('Erro ao salvar respostas:', error);
        res.status(500).json({ error: 'Erro ao salvar respostas' });
    }
};

const getRecomendacoes = async (req, res) => {
    try {
        const userId = req.user.id;
        
        const [respostas] = await pool.query('SELECT * FROM questionario_respostas WHERE user_id = ?', [userId]);
        
        if (!respostas[0]) {
            return res.json({ success: false, message: 'Responda o questionário primeiro' });

        console.error('Erro ao obter recomendações:', error);
        res.status(500).json({ error: 'Erro ao obter recomendações' });
    }

const salvarRecomendacao = async (req, res) => {
    try {
        const userId = req.user.id;

            'INSERT INTO recomendacoes_salvas (user_id, trilha_nome, similaridade) VALUES (?, ?, ?)',

    def recommend(self, vetor_usuario, top_n=5):
        trilhas_matrix = np.array([t["cursos"] for t in self.trilhas.values()])
        nomes = list(self.trilhas.keys())
        
        similaridades = cosine_similarity([vetor_usuario], trilhas_matrix)[0]
        
        recomendacoes = []
        for i, nome in enumerate(nomes):
            if similaridades[i] > 0:

                    "cursos_chave": self.trilhas[nome]["cursos_chave"]
                })
        
        recomendacoes.sort(key=lambda x: x["similaridade"], reverse=True)
        return recomendacoes[:top_n]

```javascript
const API_URL = 'http://localhost:3000/api';

async function fazerLogin(email, password) {
    try {
        const response = await fetch(`${API_URL}/auth/login`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ email, password })
        });
        const data = await response.json();
        
        if (response.ok && data.success) {
            localStorage.setItem('token', data.token);
            localStorage.setItem('user', JSON.stringify(data.user));
            return { success: true };
        }
        return { success: false, error: data.error };
    } catch (error) {
        return { success: false, error: 'Erro de conexão' };
    }
}

async function fazerCadastro(name, email, password) {
    try {
        const response = await fetch(`${API_URL}/auth/register`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ name, email, password })
        });
        const data = await response.json();
        
        if (response.ok && data.success) {
            return { success: true };
        }
        return { success: false, error: data.error };
    } catch (error) {
        return { success: false, error: 'Erro de conexão' };
    }
}

function sair() {
    localStorage.removeItem('token');
    localStorage.removeItem('user');
    window.location.href = 'login.html';
}

function isLogado() {
    const token = localStorage.getItem('token');
    if (!token) return false;
    
    try {
        const payload = JSON.parse(atob(token.split('.')[1]));
        return payload.exp > Date.now() / 1000;
    } catch {
        return false;
    }
}

                        <strong>📊 Seu nível:</strong> ${nivelTexto} (${data.nivel_usuario}/3)
                    </div>
                    ${data.recomendacoes.map(rec => `
                        <div class="recomendacao-card">
                            <h3>${rec.trilha}</h3>
                            <div style="display: flex; gap: 15px; margin: 10px 0; font-size: 0.85rem; color: #666;">
                                <span>📚 ${rec.duracao}</span>
                                <span>🎯 Nível ${rec.nivel_medio < 1.5 ? 'Iniciante' : rec.nivel_medio < 2.5 ? 'Intermediário' : 'Avançado'}</span>
                            </div>
                            <p>${rec.descricao}</p>
                            <div style="margin-top: 10px;">
                                <span class="similaridade">${rec.similaridade}% compatível</span>
                            </div>
                        </div>
                    `).join('')}
                    <div style="text-align: center; margin-top: 30px;">
                        <button class="btn-primary" onclick="reiniciar()">↻ Refazer Questionário</button>
                        <button class="btn-secondary" onclick="window.location.href='dashboard.html'">📊 Dashboard</button>
                    </div>
                </div>
            `;
            resultsDiv.style.display = 'block';
        }
        
        function reiniciar() {
            respostas = {};
            perguntaAtual = 0;
            document.getElementById('results').style.display = 'none';
            document.getElementById('questionarioContainer').style.display = 'block';
            document.getElementById('progressFill').style.width = '0%';
            document.querySelectorAll('.option').forEach(opt => opt.classList.remove('selected'));
            mostrarPergunta(0);
        }
        
        carregarPerguntas();
    </script>
</body>
</html>
```
